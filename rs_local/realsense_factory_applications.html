실시간 높이 측정 (제품 통과 시)

현재 프레임과 배경의 차분으로 높이를 산출합니다. 상위 95 퍼센타일로 노이즈를 제거합니다.

# 실시간 프레임에서 높이 산출
def measure_height(depth_now, depth_bg, depth_scale, threshold_mm=10):
    # 배경 대비 높이 차이 (mm 단위)
    diff_mm = (depth_bg - depth_now.astype(np.float64)) * depth_scale * 1000

    # threshold 이상 솟아오른 영역 = 제품이 있는 곳
    mask = diff_mm > threshold_mm  # 10mm 이상 차이나는 픽셀만

    if np.sum(mask) < 100:        # 유효 픽셀 100개 미만 = 제품 없음
        return None, None

    # 상위 95% 값 = 제품 꼭대기 높이 (노이즈 제거)
    height_mm = float(np.percentile(diff_mm[mask], 95))
    return round(height_mm, 1), mask
3
합격/불합격 판정
# 규격 검사: 50mm ± 2mm
SPEC_HEIGHT = 50.0  # 기준 높이 (mm)
TOLERANCE = 2.0     # 허용 오차 (mm)

height, mask = measure_height(depth_now, depth_bg, depth_scale)
if height is not None:
    deviation = abs(height - SPEC_HEIGHT)  # 기준 대비 편차
    if deviation <= TOLERANCE:
        result = "✅ 합격"     # 편차 2mm 이내
    else:
        result = "❌ 불합격"    # 편차 2mm 초과
    print(f"높이: {height:.1f}mm  편차: {deviation:.1f}mm  → {result}")

왜 TURBO 컬러맵이 결함 감지에 좋은가?
JET은 초록→노랑 구간에서 색 변화가 급격해서 0.3mm 차이와 0.8mm 차이가 비슷해 보입니다.
TURBO는 모든 구간에서 색 변화가 균일해서 0.3mm = 약한 노랑, 0.8mm = 진한 빨강으로 심각도를 직감할 수 있습니다.

검사품과 비교 → 편차 히트맵 생성
# 검사품 Depth와 양품의 편차를 TURBO 컬러맵으로 시각화
def detect_defect(depth_now, depth_golden, depth_scale, threshold_mm=0.5):
    # 양품 대비 편차 (mm) — 양수 = 튀어나옴, 음수 = 찍힘
    diff_mm = np.abs(
        depth_now.astype(np.float64) - depth_golden
    ) * depth_scale * 1000  # mm 변환

    # 0.5mm 이상 편차 = 결함 후보
    defect_mask = diff_mm > threshold_mm

    # 편차를 0~255로 정규화 → TURBO 컬러맵 적용
    heatmap = cv2.normalize(diff_mm, None, 0, 255, cv2.NORM_MINMAX)
    heatmap = heatmap.astype(np.uint8)
    heatmap_color = cv2.applyColorMap(heatmap, cv2.COLORMAP_TURBO)  # ← TURBO!

    # 결함 면적 비율 (%) = 결함 픽셀 수 / 전체 픽셀 수
    defect_pct = np.sum(defect_mask) / defect_mask.size * 100
    return defect_mask, defect_pct, heatmap_color

# 결함 판정: 결함 면적이 1% 이상이면 불합격
mask, pct, heatmap = detect_defect(depth_now, depth_golden, depth_scale)

if pct > 1.0:  # 결함 면적 1% 초과
    # 결함 영역의 외곽선(contour)을 찾아서 빨간 박스 표시
    contours, _ = cv2.findContours(
        mask.astype(np.uint8), cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
    )

            x, y, w, h = cv2.boundingRect(cnt)  # 결함 박스
            cv2.rectangle(color_image, (x,y), (x+w,y+h), (0,0,255), 2)
            cv2.putText(color_image, "DEFECT", (x,y-5),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0,0,255), 2)

채움률(%) 산출

ROI 내 각 픽셀의 높이를 적분하여 체적을 구하고, 최대 체적 대비 비율을 계산합니다.

# 적재량 측정 — ROI 내 Depth 차분 적분
def measure_fill_level(depth_now, depth_empty, depth_scale, roi, max_h_mm):
    x1, y1, x2, y2 = roi
    # ROI 내 빈 팔레트 대비 높이 (mm)
    roi_empty = depth_empty[y1:y2, x1:x2]
    roi_now = depth_now[y1:y2, x1:x2].astype(np.float64)
    height_mm = (roi_empty - roi_now) * depth_scale * 1000  # mm 변환

    # 0보다 큰 곳만 = 물건이 있는 곳
    height_mm = np.clip(height_mm, 0, max_h_mm)  # 최대 높이 제한

    # 평균 높이 / 최대 높이 = 채움률 (%)
    avg_height = float(np.mean(height_mm))  # 전체 ROI의 평균 높이
    fill_pct = (avg_height / max_h_mm) * 100
    return round(fill_pct, 1), round(avg_height, 1)

if fill_pct >= 80:
    status = "✅ 출하 가능"        # 80% 이상 → 충분
elif fill_pct >= 30:
    status = "⚠️ 추가 적재 필요"    # 30~80% → 부족
else:
    status = "📭 거의 비어있음"      # 30% 미만

print(f"채움률: {fill_pct:.1f}%  평균높이: {avg_h:.1f}mm  → {status}")
